  public String getTestSuiteClassName()
  {
    String result = getTestSuiteClass();
    if (isBlank(result))
    {
      result = getModelName() + "AllTests";  
    }
    else
    {
      int index = result.lastIndexOf(".");
      if (index != -1)
      {
        result = result.substring(index + 1);
      }
    }
    return result;
  }
