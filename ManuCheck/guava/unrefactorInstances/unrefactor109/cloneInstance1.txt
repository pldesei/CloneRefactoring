    @Override public boolean remove(@Nullable Object o) {
        if (o == null) return false;
        final E[] items = this.items;
        final Monitor monitor = this.monitor;
        monitor.enter();
        try {
            int i = takeIndex;
            int k = 0;
            for (;;) {
                if (k++ >= count)
                    return false;
                if (o.equals(items[i])) {
                    removeAt(i);
                    return true;
                }
                i = inc(i);
            }
        } finally {
            monitor.leave();
        }
    }
