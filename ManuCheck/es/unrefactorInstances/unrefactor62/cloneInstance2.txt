        public List<Long> getValues() {
            if (!listLoaded) {
                final int numValues = values.setDocument(docId);
                list.offset = 0;
                list.grow(numValues);
                list.length = numValues;
                for (int i = 0; i < numValues; i++) {
                    list.values[i] = values.nextValue();
                }
                listLoaded = true;
            }
            return list;
        }
