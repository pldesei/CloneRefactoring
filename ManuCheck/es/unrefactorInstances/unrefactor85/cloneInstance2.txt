    @Override public void writeTo(StreamOutput out) throws IOException {
        out.writeVLong(timestamp);
        if (cpu == null) {
            out.writeBoolean(false);
        } else {
            out.writeBoolean(true);
            cpu.writeTo(out);
        }
        if (mem == null) {
            out.writeBoolean(false);
        } else {
            out.writeBoolean(true);
            mem.writeTo(out);
        }
        if (fd == null) {
            out.writeBoolean(false);
        } else {
            out.writeBoolean(true);
            fd.writeTo(out);
        }
    }
