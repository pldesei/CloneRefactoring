    public void afterIndexClosed(Index index) {
        for (Listener listener : listeners) {
            try {
                listener.afterIndexClosed(index);
            } catch (Throwable t) {
                logger.warn("[{}] failed to invoke after index closed callback", t, index.name());
            }
        }
    }
