    public SizeValue estimatedFlushableMemorySize() {
        long bytes = -1;
        for (IndexShardStatus shard : this) {
            if (shard.estimatedFlushableMemorySize() != null) {
                if (bytes == -1) {
                    bytes = 0;
                }
                bytes += shard.estimatedFlushableMemorySize().bytes();
            }
        }
        if (bytes == -1) {
            return null;
        }
        return new SizeValue(bytes);
    }
