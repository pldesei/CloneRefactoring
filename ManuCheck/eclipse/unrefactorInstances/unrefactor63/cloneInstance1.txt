public final boolean areParametersEqual(MethodBinding method) {
    TypeBinding[] args = method.parameters;
    if (parameters == args)
        return true;

    int length = parameters.length;
    if (length != args.length)
        return false;
    
    for (int i = 0; i < length; i++)
        if (parameters[i] != args[i])
            return false;
    return true;
}
