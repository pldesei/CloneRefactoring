public boolean containsKey(int key) {
    int index = key % valueTable.length;
    int currentKey;
    while ((currentKey = keyTable[index]) != 0) {
        if (currentKey == key)
            return true;
        index = (index + 1) % keyTable.length;
    }
    return false;
}
