  Packed16ThreeBlocks(DataInput in, int valueCount) throws IOException {
    this(valueCount);
    for (int i = 0; i < 3 * valueCount; ++i) {
      blocks[i] = in.readShort();
    }
    final int mod = blocks.length % 4;
    if (mod != 0) {
      for (int i = mod; i < 4; ++i) {
         in.readShort();
      }
    }
  }
