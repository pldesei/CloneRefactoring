  void doIterate1(BitSet a, OpenBitSet b) {
    int aa=-1,bb=-1;
    OpenBitSetIterator iterator = new OpenBitSetIterator(b);
    do {
      aa = a.nextSetBit(aa+1);
      if (rand.nextBoolean())
        iterator.next();
      else
        iterator.skipTo(bb+1);
      bb = iterator.doc();
      assertEquals(aa,bb);
    } while (aa>=0);
  }
