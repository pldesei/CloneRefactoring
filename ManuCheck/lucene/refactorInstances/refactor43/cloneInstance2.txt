  Direct32(DataInput in, int valueCount) throws IOException {
    this(valueCount);
    for (int i = 0; i < valueCount; ++i) {
      values[i] = in.readInt();
    }
    final int mod = valueCount % 2;
    if (mod != 0) {
      for (int i = mod; i < 2; ++i) {
        in.readInt();
      }
    }
  }
