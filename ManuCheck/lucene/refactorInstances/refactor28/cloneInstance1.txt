  public DocsEnum docs(Bits skipDocs, BytesRef text, DocsEnum reuse) throws IOException {

    if (termsEnum.seek(text) == TermsEnum.SeekStatus.FOUND) {
      return termsEnum.docs(skipDocs, reuse);

      return null;
