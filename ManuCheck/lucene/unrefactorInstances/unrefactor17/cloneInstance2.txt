    public int nextDoc() {
      upto += 2;
      if (liveDocs == null) {
        if (upto < postings.length) {
          return postings[upto];
        }
      } else {
        while (upto < postings.length) {
          if (liveDocs.get(postings[upto])) {
            return postings[upto];
          }
          upto += 2;
        }
      }
      return NO_MORE_DOCS;
    }
