    private static CommandLauncher extractLauncher(String referenceName,
                                                   Project project) {
        CommandLauncher launcher = null;
        if (project != null) {
            launcher = (CommandLauncher) project.getReference(referenceName);
        }

        if (launcher == null) {
            launcher = getSystemLauncher(referenceName);
        }
        return launcher;
    }
