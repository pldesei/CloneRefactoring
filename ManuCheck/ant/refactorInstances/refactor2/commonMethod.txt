    protected final String readLine() throws IOException {
        int ch = in.read();
        String line = (ch == -1) ? null : "";
        while (ch != -1) {
            line += (char) ch;
            if (ch == '\n') {
                break;
            }
            ch = in.read();
        }
        return line;
    }
