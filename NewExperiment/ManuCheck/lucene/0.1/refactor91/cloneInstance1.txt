  Direct8(DataInput in, int valueCount) throws IOException {
    this(valueCount);
    for (int i = 0; i < valueCount; ++i) {
      values[i] = in.readByte();
    }
    final int mod = valueCount % 8;
    if (mod != 0) {
      for (int i = mod; i < 8; ++i) {
        in.readByte();
      }
    }
  }
