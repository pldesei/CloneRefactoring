    /**
     * Remember a position in a string buffer.
     *
     * @param sb The <code>StringBuilder</code> to mark.
     * @return The current buffer position if non-null.
     */
    public static int sbMark(StringBuilder sb) {
        return (sb == null) ? -1 : sb.length();
