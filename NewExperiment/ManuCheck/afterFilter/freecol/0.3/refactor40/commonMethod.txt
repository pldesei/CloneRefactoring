	/**
	 * Removes all references to this object.
	 */
	public void dispose() {
		if (isCarrier()) {
			unitContainer.dispose();
			goodsContainer.dispose();
		}

		if (location != null) {
			location.remove(this);
		}

		setIndianSettlement(null);

		getOwner().invalidateCanSeeTiles();

		super.dispose();
	}
