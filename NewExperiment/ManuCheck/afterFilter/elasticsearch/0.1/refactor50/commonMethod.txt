    public void writeBytesHolder(@Nullable BytesHolder bytes) throws IOException {
        if (bytes == null) {
            writeVInt(0);
        } else {
            writeVInt(bytes.length());
            writeBytes(bytes.bytes(), bytes.offset(), bytes.length());
        }
    }
