    protected IRubyObject[] convertToRubyArray(ThreadContext context, IRubyObject[] args) {
        return (args.length == 0) ? context.getRuntime().getSingleNilArray()
                                  : new IRubyObject[] {context.getRuntime().newArrayNoCopy(args)};
    }
