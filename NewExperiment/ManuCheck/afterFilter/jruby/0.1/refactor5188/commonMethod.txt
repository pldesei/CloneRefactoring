    private static void writePemPlain(BufferedWriter out, String pemHeader, byte[] encoding) throws IOException {
        out.write(BEF_G + pemHeader + AFT);
        out.newLine();
        writeEncoded(out, encoding);
        out.write(BEF_E + pemHeader + AFT);
        out.newLine();
        out.flush();
    }
