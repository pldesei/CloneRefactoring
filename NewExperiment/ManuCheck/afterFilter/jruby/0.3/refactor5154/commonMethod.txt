    public boolean usesLocalVariable(Variable v) {
        if (_usedLocalVars.contains(v)) {
            return true;
        }
        else {
            for (IRClosure cl: getScope().getClosures()) {
                if (cl.getCFG().usesLocalVariable(v))
                    return true;
            }

            return false;
        }
    }
