    private void determineExpressionState() {
        switch (lex_state) {
        case EXPR_FNAME: case EXPR_DOT:
            lex_state = LexState.EXPR_ARG;
            break;
        default:
            lex_state = LexState.EXPR_BEG;
            break;
        }
    }
