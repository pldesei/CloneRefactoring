  static ByteSource asByteSource(
      final InputSupplier<? extends InputStream> supplier) {
    checkNotNull(supplier);
    return new ByteSource() {
      @Override
      public InputStream openStream() throws IOException {
        return supplier.getInput();
      }
    };
  }
