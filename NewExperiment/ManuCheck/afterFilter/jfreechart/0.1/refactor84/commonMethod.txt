    /**
     * Returns <code>true</code> if this orientation is <code>HORIZONTAL</code>,
     * and <code>false</code> otherwise.  
     * 
     * @return A boolean.
     * 
     * @since 1.0.18
     */
    public boolean isHorizontal() {
        return this.equals(PlotOrientation.HORIZONTAL);
    }
