	/**
	 * Resize the pool contents
	 */
	private final void resizePoolContents(int minimalSize) {
		final int contentsLength = this.contents.length;
		System.arraycopy(
			this.contents,
			0,
			(this.contents = new byte[contentsLength + Math.max(INCREMENT_SIZE, minimalSize)]),
			0,
			contentsLength);
	}
