  private boolean checkStddevs(ArrayList<Double> list1, ArrayList<Double> list2) {
    for (int i = 0; i<list1.size(); i++) {
      if ((Math.abs(list1.get(i)-list2.get(i))<.00000000001) == false) {
        return false;
      }
    }
    return true;
  }
