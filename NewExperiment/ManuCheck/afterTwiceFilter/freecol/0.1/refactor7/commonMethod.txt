    /**
    * Checks if this unit is visible to the given player.
    */
    public boolean isVisibleTo(Player player) {
        return (getTile() != null && player.canSee(getTile()) && (getTile().getSettlement() == null
                || getTile().getSettlement().getOwner() == player));
    }
