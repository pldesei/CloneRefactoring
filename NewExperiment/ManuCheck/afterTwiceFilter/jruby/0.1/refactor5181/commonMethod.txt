    private void pushBBOnStack(Stack<BasicBlock> stack, BitSet bbSet, BasicBlock bb) {
        if (!bbSet.get(bb.getID())) {
            // System.out.println("pushing " + bb);
            stack.push(bb);
            bbSet.set(bb.getID());
        }
    }
