    private int greaterThan() throws IOException {
        determineExpressionState();

        int c = src.read();

        switch (c) {
        case '=':
            yaccValue = new Token(">=", getPosition());
            return Tokens.tGEQ;
        case '>':
            if ((c = src.read()) == '=') {
                setState(LexState.EXPR_BEG);
                yaccValue = new Token(">>=", getPosition());
                return Tokens.tOP_ASGN;
            }
            src.unread(c);
            yaccValue = new Token(">>", getPosition());
            
            return Tokens.tRSHFT;
        default:
            src.unread(c);
            
            yaccValue = new Token(">", getPosition());
            return Tokens.tGT;
        }
    }
