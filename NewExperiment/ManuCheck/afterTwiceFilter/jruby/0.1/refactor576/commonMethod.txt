    private final RubyMatchData matchdata(Ruby runtime, String str, Matcher mat, boolean utf8) { 
        if (utf8) {
            return new RubyMatchData.JavaString(runtime, str, mat);
        } else {
            return new RubyMatchData.RString(runtime, this, mat);
        }        
    }
