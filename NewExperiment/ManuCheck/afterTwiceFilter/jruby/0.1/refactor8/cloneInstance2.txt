    public RubyObject times() {
        RubyNumeric i = RubyFixnum.newFixnum(getRuby(), 0);
        while (true) {
            if (i.funcall(getRuby().intern("<"), this).isFalse()) {
                break;
            }
            getRuby().yield(i);
            i = (RubyNumeric) i.funcall(getRuby().intern("+"), RubyFixnum.newFixnum(getRuby(), 1));
        }
        return this;
    }
