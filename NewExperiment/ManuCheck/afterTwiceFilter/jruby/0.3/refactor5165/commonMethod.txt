    private Variable getValueInTemporaryVariable(IRScope s, Operand val) {
        if ((val != null) && (val instanceof TemporaryVariable)) {
            return (Variable)val;
        }
        else {
            Variable tmpVar = s.getNewTemporaryVariable();
            s.addInstr(new CopyInstr(tmpVar, val));
            return tmpVar;
        }
    }
