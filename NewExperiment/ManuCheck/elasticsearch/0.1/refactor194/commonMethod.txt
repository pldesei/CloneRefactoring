    public List<Cell> getHeadersFromNames(List<String> headerNames) {
        List<Cell> hdrs = new ArrayList<Cell>();
        for (String hdrToFind : headerNames) {
            for (Cell header : headers) {
                if (((String) header.value).equalsIgnoreCase(hdrToFind)) {
                    hdrs.add(header);
                }
            }
        }
        return hdrs;
    }
