  public long writeFrom(InputStream input) throws IOException {
    checkNotNull(input);

    Closer closer = Closer.create();
    try {
      OutputStream out = closer.add(openStream());
      return ByteStreams.copy(input, out);
    } catch (Throwable e) {
      throw closer.rethrow(e, IOException.class);
    } finally {
      closer.close();
    }
  }
