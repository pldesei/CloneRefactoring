  public byte[] read() throws IOException {
    Closer closer = Closer.create();
    try {
      InputStream in = closer.add(openStream());
      return ByteStreams.toByteArray(in);
    } catch (Throwable e) {
      throw closer.rethrow(e, IOException.class);
    } finally {
      closer.close();
    }
  }
