  public String read() throws IOException {
    Closer closer = Closer.create();
    try {
      Reader reader = closer.add(openStream());
      return CharStreams.toString(reader);
    } catch (Throwable e) {
      throw closer.rethrow(e, IOException.class);
    } finally {
      closer.close();
    }
  }
