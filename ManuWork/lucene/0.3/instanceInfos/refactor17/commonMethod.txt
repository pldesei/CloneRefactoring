    public final boolean incrementToken() throws IOException
    {
      if (input.incrementToken()) {
        String term = termAtt.term();
        String s = stemmer.stem(term);
        if (s != null && !s.equals(term))
          termAtt.setTermBuffer(s);
        return true;
      } else {

      }
