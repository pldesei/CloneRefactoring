public void testChangeMethodVisibility() throws JavaModelException {
    this.workingCopy.getBuffer().setContents(
        "package p1;\n" +
        "import p2.*;\n" +
        "public class X {\n" +
        "  private void foo() {\n" +
        "  }\n" +
        "}");
    this.workingCopy.reconcile(false, null);
    assertDeltas(
        "Unexpected delta", 
        "X[*]: {CHILDREN | FINE GRAINED}\n" +
        "    foo()[*]: {MODIFIERS CHANGED}"
    );
}
